pipeline {
    agent any

    environment {
        SSH_CRED_ID = 'ccb-deploy-ssh'
        HOST_MAIN   = ''
        HOST_2 = ''
    }

    stages {

        stage('STOP WebLogic') {
            steps {
                script {
                    echo "üõë WebLogic STOP (${HOST_MAIN})"

                    withCredentials([sshUserPrivateKey(
                        credentialsId: SSH_CRED_ID,
                        keyFileVariable: 'SSH_KEY',
                        usernameVariable: 'SSH_USER'
                    )]) {
                        sh """
                          ssh -i "\$SSH_KEY" -o StrictHostKeyChecking=no \
                          \$SSH_USER@${HOST_MAIN} "
                            cd /appdata/restart
                            sh stop.sh
                          "
                        """
                    }
                }
            }
        }
        stage('renamefolder (MAIN)') {
            steps {
                script {
                    echo "üìÅ renamefolder2.sh √ßalƒ±≈üƒ±yor (MAIN)..."

                    withCredentials([sshUserPrivateKey(
                        credentialsId: SSH_CRED_ID,
                        keyFileVariable: 'SSH_KEY',
                        usernameVariable: 'SSH_USER'
                    )]) {

                        sh """
                            ssh -i "\$SSH_KEY" -o StrictHostKeyChecking=no \$SSH_USER@${HOST_MAIN} "
                                cd /appdata/scripts
                                sh renamefolder2.sh
                            "
                        """
                    }
                }
            }
        }

        stage('renamefolder (2)') {
            steps {
                script {
                    echo "üìÅ renamefolder2.sh √ßalƒ±≈üƒ±yor (2)..."

                    withCredentials([sshUserPrivateKey(
                        credentialsId: SSH_CRED_ID,
                        keyFileVariable: 'SSH_KEY',
                        usernameVariable: 'SSH_USER'
                    )]) {

                        sh """
                            ssh -i "\$SSH_KEY" -o StrictHostKeyChecking=no \$SSH_USER@${HOST_2} "
                                cd /appdata/scripts
                                sh renamefolder2.sh
                            "
                        """
                    }
                }
            }
        }
        stage('Manual Confirm') {
            steps {
                input "STOP tamamlandƒ±. START i√ßin Proceed."
            }
        }
   
        stage('START WebLogic') {
            steps {
                script {
                    echo "üöÄ WebLogic START (${HOST_MAIN})"

                    withCredentials([sshUserPrivateKey(
                        credentialsId: SSH_CRED_ID,
                        keyFileVariable: 'SSH_KEY',
                        usernameVariable: 'SSH_USER'
                    )]) {
                        sh """
                          ssh -i "\$SSH_KEY" -o StrictHostKeyChecking=no \
                          \$SSH_USER@${HOST_MAIN} "
                            cd /appdata/restart
                            sh start.sh
                          "
                        """
                    }
                }
            }
        }
    }
}
